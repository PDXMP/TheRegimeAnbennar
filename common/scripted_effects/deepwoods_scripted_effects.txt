# Annexes all wood elf vassals, changes player to cyranvar, locks admin, updates MT
form_cyranvar = {
	change_tag = I45
	hidden_effect = { 
		restore_country_name = yes
		every_subject_country = {
			limit = {
				primary_culture = wood_elf
				capital_scope = { superregion = deepwoods_superregion }
				NOT = { has_country_flag = I28_no_inherit } #drennen event 11
			}
			I45 = { inherit = PREV }
		}

		# Salla Myna modifiers that are removed upon forming Cyranvar.
		remove_country_modifier = I27_yfelorr_share_mod
		remove_country_modifier = I27_trusted_ceanna_mod
	}
	swap_non_generic_missions = yes
	remove_non_electors_emperors_from_empire_effect = yes
	if = {
		limit = {
			NOT = { government_rank = 2 }
		}
		set_government_rank = 2
	}
	custom_tooltip = unite_deepwoods_tooltip
	country_event = { id = ideagroups.1 } #Swap Ideas
	lock_racial_admin = yes
	add_country_modifier = {
		name = "centralization_modifier"
		duration = 7300
	}
}

spiked_log_upgrade_foreign_ideas = {
	if = {
		limit = {
			NOT = { has_estate_privilege = estate_monstrous_tribes_the_fallen_veil }
			NOT = { has_estate_privilege = estate_monstrous_tribes_the_open_forest }
		}
		set_estate_privilege = estate_monstrous_tribes_the_fallen_veil
	}
	if = {
		limit = {
			has_country_flag = spiked_log_upgrade_foreign_ideas_flag_1
			has_country_flag = spiked_log_upgrade_foreign_ideas_flag_2
			has_country_flag = spiked_log_upgrade_foreign_ideas_flag_3
			has_country_flag = spiked_log_upgrade_foreign_ideas_flag_4
		}
		remove_estate_privilege = estate_monstrous_tribes_the_fallen_veil
		set_estate_privilege = estate_monstrous_tribes_the_open_forest
		clr_country_flag = spiked_log_upgrade_foreign_ideas_flag_1
		clr_country_flag = spiked_log_upgrade_foreign_ideas_flag_2
		clr_country_flag = spiked_log_upgrade_foreign_ideas_flag_3
		clr_country_flag = spiked_log_upgrade_foreign_ideas_flag_4
	}
	else = {
		custom_tooltip = spiked_log_upgrade_foreign_ideas_tt
	}
}
