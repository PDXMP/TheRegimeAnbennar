S98_hero_worship_conquest_effect = {
	owner = {
		if = {
			limit = {
				NOT = {
					has_ruler_modifier = S98_hero_worship_ruler_modifier_conquest
					has_ruler_modifier = S98_hero_worship_ruler_modifier_building
					has_ruler_modifier = S98_hero_worship_ruler_modifier_dev
				}
				check_variable = {
					which = s98heroWorshipConquest
					which = s98heroWorshipTotalConquest
				}
			}
			add_ruler_modifier = {
				name = S98_hero_worship_ruler_modifier_conquest
			}
			set_variable = {
				which = s98heroWorshipConquest
				value = 0
			}
			if = { 
				limit = {
					NOT = {
						check_variable = {
							which = s98heroWorshipTotalConquest
							value = 20
						}
					}
				}
				change_variable = {
					which = s98heroWorshipTotalConquest
					value = 5
				}
			}
		}
	}
}

S98_change_province_effect = {
	if = {
		limit = { 
			owner = {
				OR = { 
					is_or_was_tag = { tag = S98 }
					overlord = {
						is_or_was_tag = { tag = S98 }
					}
				} 
			}
		}
		if = {
			limit = {
				owner = {
					OR = {
						has_country_flag = S98_settling_among_jiromji_finished
						overlord = {
							has_country_flag = S98_settling_among_jiromji_finished
						}
					}
				}
				culture = jiromji
			}
			add_halfling_minority_size_effect = yes
			add_halfling_minority_size_effect = yes
		}
		if = {
			limit = {
				owner = {
					has_country_flag = S98_settling_among_jiromji_finished
				}
				culture_group = west_sarhaly
			}
			add_province_modifier = {
				name = S98_religious_tolerance
				duration = -1
			}
		}
		if = {
			limit = {
				owner = {
					has_reform = bibyobi_dynasties_reform
				}
				has_human_majority_trigger = yes
			}
			add_province_modifier = {
				name = S98_human_tolerance
				duration = -1
				hidden = yes
			}
		}
		if = {
			limit = {
				owner = {
					OR = {
						has_country_flag = S98_the_kaino_finished
						overlord = {
							has_country_flag = S98_the_kaino_finished
						}
					}
				}
				culture = kaino
			}
			add_province_modifier = {
				name = S98_religious_tolerance
				duration = -1
			}
		}
		if = {
			limit = {
				owner = {
					OR = {
						has_country_flag = S98_fighting_the_khet_finished
						overlord = {
							has_country_flag = S98_fighting_the_khet_finished
						}
					}
				}
				region = horashesh_region
				religion = noukahi_pantheon				
			}
			add_province_modifier = {
				name = S98_religious_tolerance
				duration = -1
			}
			add_halfling_minority_size_effect = yes
			add_halfling_minority_size_effect = yes
		}

		if = {
			limit = {
				owner = { has_country_flag = S98_hero_worship_active }
			}
			owner = {
				change_variable = {
					which = s98heroWorshipConquest
					value = 1
				}
			}
			S98_hero_worship_conquest_effect = yes
		}

		if = {
			limit = {
				owner = {
					has_country_flag = S98_rest_of_fangaulans_finished
				}
				OR = {
					has_terrain = grasslands
					has_terrain = steppe
					has_terrain = savannah
				}
			}
			add_province_modifier = {
				name = S98_fangaulan_planes
				duration = -1
			}
		}

		if = {
			limit = {
				owner = { has_country_flag = S98_last_of_khetists_finished }
				religion_group = khetist
			}
			set_province_flag = S98_conquered_khetists
		}
		if = {
			limit = {
				owner = {
					OR = {
						has_country_modifier = S98_khetist_missionary_strength_on
						overlord = {
							has_country_modifier = S98_khetist_missionary_strength_on
						}
					}
				}
				region = horashesh_region
				religion_group = khetist
				owner = {
					NOT = {
						religion_group = khetist
					}
				}
			}
			add_province_modifier = {
				name = S98_khetist_missionary_strength
				duration = 3650
			}
		}
	}
}
