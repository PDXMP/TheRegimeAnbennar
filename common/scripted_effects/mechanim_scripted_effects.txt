	}

		limit = {
			check_variable = {
				which = mechanim_progress_level
				value = 2

		add_dip_power = 25
		add_mil_power = 25

			check_variable = {
				which = mechanim_progress_level
				value = 1
			}
			NOT = {
				check_variable = {
					which = mechanim_progress_level

				}

			NOT = {

					has_country_flag = unlocked_manago_poku_lesson_discovery
