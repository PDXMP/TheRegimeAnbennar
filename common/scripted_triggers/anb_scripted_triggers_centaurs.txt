all_nzoc_province_has_trigger = { #Province Scope
	AND = {
		$trigger$
	}
	all_neighbor_province = {
		$trigger$
	}
	if = { #Golden Tremor
		limit = { has_saved_global_event_target = I57_permanent_capital_target }
		event_target:I57_permanent_capital_target = {
			$trigger$
		}
	}
}

any_nzoc_province_has_trigger = { #Province Scope
	OR = {
		AND = {
			$trigger$
		}
		any_neighbor_province = {
			$trigger$
		}
		if = { #Golden Tremor
			limit = { has_saved_global_event_target = I57_permanent_capital_target }
			event_target:I57_permanent_capital_target = {
				$trigger$
			}
		}
		else = { hidden_trigger = { always = no } }
	}
}

amount_nzoc_province_has_trigger = { #Country Scope
	num_of_owned_provinces_with = {
		value = $amount$
		OR = {
			province_id = ROOT
			any_neighbor_province = { province_id = ROOT }
			if = { #Golden Tremor
				limit = { has_saved_global_event_target = I57_permanent_capital_target }
				province_id = event_target:I57_permanent_capital_target
			}
			else = { hidden_trigger = { always = no } }
		}
		$trigger$
	}
}
