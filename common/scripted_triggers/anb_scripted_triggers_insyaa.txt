
tag_can_use_mechanim_ward_conduit = {
	# Mechanim formables
	OR = {
		tag = M22 # Kiretua
		tag = M23 # Rangitema
		tag = M24 # Anahira
		tag = M25 # Pa-Aroha
		tag = M26 # Rakauma
	}
}

province_can_get_mechanim_ward_conduit = {
	if = {
		limit = { owner = { tag = M22 } } # Kiretua
		owner = { tag = M22 }
		OR = {
			area = titanoflora_lakelands_5_area
			area = titanoflora_lakelands_6_area
		}
	}
	else_if = {
		limit = { owner = { tag = M23 } } # Rangitema
		owner = { tag = M23 }
		OR = {
			area = titanoflora_lakelands_10_area
			area = titanoflora_lakelands_11_area
			area = titanoflora_lakelands_12_area
			area = titanoflora_lakelands_13_area
		}
	}
	else_if = {
		limit = { owner = { tag = M24 } } # Anahira
		owner = { tag = M24 }
		OR = {
			area = titanoflora_riverlands_11_area
			area = titanoflora_riverlands_13_area
			province_id = 3496
			province_id = 3331
			province_id = 3495
		}
	}
	else_if = {
		limit = { owner = { tag = M25 } } # Pa-Aroha
		owner = { tag = M25 } # Pa-Aroha
		OR = {
			area = titanoflora_riverlands_16_area
			area = titanoflora_riverlands_17_area
			province_id = 3183
			province_id = 3181
			province_id = 3322
			province_id = 3323
		}
	}
	else_if = {
		limit = { owner = { tag = M26 } } # Rakauma
		owner = { tag = M26 }
		OR = {
			area = turuputai_area
			area = manuata_area
			province_id = 3159
			province_id = 3560
		}
	}
	else = {
		# unreachable because this is locked behind tag_can_use_mechanim_ward_conduit
		always = no
	}
}
