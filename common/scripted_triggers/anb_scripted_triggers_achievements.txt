get_colored_achievement_for_page = {
	OR = {
		AND = {
			has_country_flag = ach_$page_1_ach$_complete_perma
			has_country_flag = achievements_page_1_opened
		}
		AND = {
			has_country_flag = ach_$page_2_ach$_complete_perma
			has_country_flag = achievements_page_2_opened
		}
		AND = {
			has_country_flag = ach_$page_3_ach$_complete_perma
			has_country_flag = achievements_page_3_opened
		}
		[[small_resolution]
			AND = {
				has_country_flag = ach_$page_4_ach$_complete_perma
				has_country_flag = achievements_page_4_opened
			}
			AND = {
				has_country_flag = ach_$page_5_ach$_complete_perma
				has_country_flag = achievements_page_5_opened
			}
			AND = {
				has_country_flag = ach_$page_6_ach$_complete_perma
				has_country_flag = achievements_page_6_opened
			}
		]
	}
}

get_grayed_achievement_for_page = {
	OR = {
		AND = {
			NOT = { has_country_flag = ach_$page_1_ach$_complete_perma }
			has_country_flag = achievements_page_1_opened
		}
		AND = {
			NOT = { has_country_flag = ach_$page_2_ach$_complete_perma }
			has_country_flag = achievements_page_2_opened
		}
		AND = {
			NOT = { has_country_flag = ach_$page_3_ach$_complete_perma }
			has_country_flag = achievements_page_3_opened
		}
		[[small_resolution]
			AND = {
				NOT = { has_country_flag = ach_$page_4_ach$_complete_perma }
				has_country_flag = achievements_page_4_opened
			}
			AND = {
				NOT = { has_country_flag = ach_$page_5_ach$_complete_perma }
				has_country_flag = achievements_page_5_opened
			}
			AND = {
				NOT = { has_country_flag = ach_$page_6_ach$_complete_perma }
				has_country_flag = achievements_page_6_opened
			}
		]
	}
}

get_achievement_button_triggers = {
	if = {
		limit = { has_country_flag = achievements_page_1_opened }
		get_achievement_triggers = { achievement = $page_1_ach$ }
	}
	if = {
		limit = { has_country_flag = achievements_page_2_opened }
		get_achievement_triggers = { achievement = $page_2_ach$ }
	}
	if = {
		limit = { has_country_flag = achievements_page_3_opened }
		get_achievement_triggers = { achievement = $page_3_ach$ }
	}
	[[small_resolution]
		if = {
			limit = { has_country_flag = achievements_page_4_opened }
			get_achievement_triggers = { achievement = $page_4_ach$ }
		}
		if = {
			limit = { has_country_flag = achievements_page_5_opened }
			get_achievement_triggers = { achievement = $page_5_ach$ }
		}
		if = {
			limit = { has_country_flag = achievements_page_6_opened }
			get_achievement_triggers = { achievement = $page_6_ach$ }
		}
	]
}
