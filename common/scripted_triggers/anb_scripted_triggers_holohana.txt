is_valid_holohana_move_target = {
	OR = {
		has_global_flag = holohana_rampage_global
		has_global_flag = holohana_rampage_$holohana$
		NOT = {
			has_province_modifier = mechanim_ward_conduit
		}
	}
	NOT = {
		province_group = gene_pillars
		has_province_flag = holohana_dont_go_there #Use this on province you dont want holohana to enter
		has_province_flag = holohana_is_here_flag
		area = mohanoworo_area
		has_terrain = hills
		has_terrain = mountain
		has_terrain = coastal_desert
	}
	OR = {
		superregion = hinuilands_superregion
		region = titanoflora_savanna_region
		region = titanoflora_valley_region
		#region = insyaan_mushroom_forest_region 
	}
}

is_valid_holohana_spawn = {
	is_empty = yes
	NOT = {
		province_group = gene_pillars
		has_province_flag = holohana_dont_go_there
		has_province_flag = holohana_is_here_flag
		has_province_modifier = mechanim_ward_conduit
		has_province_modifier = holohana_rumbling # don't spawn next to other holohana

		has_terrain = hills

		superregion = hinuilands_superregion
		region = titanoflora_savanna_region

}

holohana_migration_start = {
	NOT = {
		unit_in_battle = yes
	}
	6234 = {
		check_variable = {
			which = $holohana$MigrateTimer
			which = $holohana$MigrateStart
		}
		NOT = { has_global_flag = holohana_dead_$holohana$ }
	}
}
holohana_fleeing_start = {
	6234 = { has_province_flag = holohana_fleeing_$holohana$ } 
	NOT = { has_global_flag = holohana_dead_$holohana$ }
	event_target:holohana_target_$holohana$ = {
		NOT = {
			unit_in_battle = yes
		}
	}
}

holohana_neighbor_modifier = {
	OR = {
		any_neighbor_province = {
			has_province_modifier = holohana_modifier_$holohana$
		}
		any_empty_neighbor_province = {
			has_province_modifier = holohana_modifier_$holohana$
		}
	}
}
holohana_not_running_back = {
	NOT = { province_id = event_target:holohana_prev_target_$holohana$ }
	NOT = { any_empty_neighbor_province = { province_id = event_target:holohana_prev_target_$holohana$ } }
	NOT = { any_neighbor_province = { province_id = event_target:holohana_prev_target_$holohana$ } }
}
