defined_text = {
	name = GetSpiritDhenbasana
	random = no
	text = {
		localisation_key = spirit_dhenbasana_yes
		trigger = {
			ai = no
			OR = {
				AND = {
					NOT = { has_country_flag = rending_appease_flag }
					NOT = { has_global_flag = dhenbasana_suppressed }
					NOT = { has_global_flag = dhenbasana_scoured }
				}
				AND = {
					has_country_flag = rending_appease_flag
					NOT = { has_country_flag = dhenbasana_appeased }
				}
			}
		}
	}
	text = {
		localisation_key = spirit_no
		trigger = { always = yes }
	}
}
