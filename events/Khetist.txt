country_event = {
	id = khetist.2
	title = khetist.2.t
	desc = khetist.2.d
	picture = indiangfx_FAMINE_eventPicture
	
	is_triggered_only = yes

	trigger = { 
		religion_group = khetist
		fast_owns_num_pops_of_race = { race = gnollish amount = 1 }
		NOT = {
    		has_country_modifier = khetist_cow_based_festival_enjoyment
    		has_country_modifier = khetist_strict_cow_ruling
			has_country_modifier = khetist_cow_based_festival_grand_enjoyment
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = { fast_has_pop_of_race = { race = gnollish } } 
				save_event_target_as = festival_province_origin
			}
			random_owned_province = {
				limit = { NOT = { province_id = event_target:festival_province_origin } }
				save_event_target_as = festival_province_2
			}
			if = {
				limit = { has_saved_event_target = festival_province_2 }
				random_owned_province = {
					limit = {
						NOT = { province_id = event_target:festival_province_origin }
						NOT = { province_id = event_target:festival_province_2 }
					}
					save_event_target_as = festival_province_3
            	}
			}
		}
	}

	option = {
		name = khetist.2.a
		small_increase_of_gnollish_tolerance_effect = yes
		event_target:festival_province_origin = {
			add_province_modifier = { name = khetist_cow_based_festival duration = 3650 }
		}
		add_country_modifier = { name = khetist_cow_based_festival_enjoyment duration = 3650 }
	}
	option = {
		name = khetist.2.b
		small_decrease_of_gnollish_tolerance_effect = yes
		event_target:festival_province_origin = {
			add_province_modifier = { name = khetist_scrutinized_cow_based_festival duration = 3650 }
		}
		add_country_modifier = { name = khetist_strict_cow_ruling duration = 3650 }
	}
	option = {
		name = khetist.2.c
		trigger = {
			OR = {
				ruler_has_personality = tolerant_personality
				ruler_has_personality = free_thinker_personality
			}
		}
		highlight = yes
		add_adm_power = -25
		add_stability = 1
		medium_increase_of_gnollish_tolerance_effect = yes
		event_target:festival_province_origin = {
			add_province_modifier = { name = khetist_cow_based_festival duration = 3650 }
		}
		event_target:festival_province_2 = {
			add_province_modifier = { name = khetist_cow_based_festival duration = 3650 }
		}
		event_target:festival_province_3 = {
			add_province_modifier = { name = khetist_cow_based_festival duration = 3650 }
		}
		add_country_modifier = { name = khetist_cow_based_festival_grand_enjoyment duration = 3650 }
	}
}
