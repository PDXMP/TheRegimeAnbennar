	id = zaradyrra.0
	title = zaradyrra.0.title
	desc = zaradyrra.0.desc
	picture = LAND_MILITARY_eventPicture

	is_triggered_only = yes
	
	trigger = {
		has_reform = ashhana_tribe
		#Make sure this doesn't happen when you load the save game
		NOT = { has_country_flag = had_ashhana_starting_event }
	}

	immediate = {
		set_country_flag = had_ashhana_starting_event
		hidden_effect = {
			#These are the two sources of Control
			set_variable = {
				which = claw_strength
				value = 0
			}
			set_variable = {
				which = canvas_strength
				value = 0
			}
			#This is used for the gov power images
			set_variable = {
				which = canvas_minus_claw
				value = 0
			}
			# Use this to check if all Ashhana are AI. If so, they can
			# have the normal revolt
			if = {
				limit = {
					ai = no
				}
				set_global_flag = human_player_ashhana
			}
		}
	}

	option = {
		name = zaradyrra.0.a
		custom_tooltip = zaradyrra.0.a.tt
	}
}

#Disaster pulse events - shared
# For the AI weighting, in general we weight 2:1 towards increasing the strength of
# the side the AI has chosen to support. If an option is gated behind other triggers,
# those options are weighted 10:1
country_event = { #The Crazed Few
	id = zaradyrra.18
	title = zaradyrra.18.title
	desc = zaradyrra.18.desc
	picture = MYSTIC_IN_PRAYER_eventPicture

	is_triggered_only = yes
	trigger = {
		#So these events only occur every 4th monthly pulse
		NOT = {
			has_country_modifier = zaradyrra_disaster_disaster_event_month_delay
		}
		NOT = {
			has_country_flag = zaradyrra_crazed_few_happened
		}
		# Just to make sure the events happen in the 2 years between the disaster starting and revolt
		NOT = { has_country_flag = ashhana_revolt_events_enabled }
	}
	
	immediate = {
		hidden_effect = {
			#So these events only occur every 4th monthly pulse
			add_country_modifier = {
				name = zaradyrra_disaster_disaster_event_month_delay
				duration = 95 #just over 3 months
				hidden = yes
			}
			set_country_flag = zaradyrra_crazed_few_happened
			random_owned_province = {
				save_event_target_as = heretic_rebel_spawn
			}
		}
	}

	option = { #Blood begets Blood
		name = zaradyrra.18.a
		ai_chance = {
			factor = 1
			# Weight the AI towards increasing the strength of the side they support
			modifier = {
				factor = 2
				has_country_flag = ashhana_disaster_supporting_claw
			}
		}
		goto = heretic_rebel_spawn

		event_target:heretic_rebel_spawn = {
			spawn_rebels = {
				size = 1
				type = heretic_rebels
			}
		}
		ashhana_blood_curse_add_faction_strength = {
			claw_strength = 3
		}
	}

	option = { #They can be saved
		name = zaradyrra.18.b
		ai_chance = {
			factor = 1
			# Weight the AI towards increasing the strength of the side they support
			modifier = {
				factor = 2
				has_country_flag = ashhana_disaster_supporting_canvas
			}
		}

		add_legitimacy_equivalent = { amount = -10 republican_tradition = -5 }
		
		ashhana_blood_curse_add_faction_strength = {
			canvas_strength = 3
		}
	}
}

country_event = { # end the disaster after 5 years without the revolt spawning
	id = zaradyrra.203
	title = zaradyrra.203.title
	desc = zaradyrra.203.desc
	picture = AFRICAN_CHIEFTAIN_eventPicture

	is_triggered_only = yes
	
	trigger = {
		# If the revolt has already happened or the disaster is resolved, just ignore this
		NOT = { has_country_flag = ashhana_revolt_events_enabled }
		NOT = { has_country_flag = ashhana_the_agony_of_expression_complete }
	}

	after = {
		end_disaster = ashhana_the_agony_of_expression

		set_country_flag = ashhana_the_agony_of_expression_complete
		clr_country_flag = no_callaction
		remove_country_modifier = zaradyrra_disaster_disaster_event_month_delay
		#Clear all the flags we were using for the disaster events, but keep the Control event
		#ones since you still meet those Control thresholds
		clr_country_flag = ashhana_revolt_events_enabled
		clr_country_flag = ashhana_disaster_can_progress
		clr_country_flag = zaradyrra_crazed_few_happened
		clr_country_flag = zaradyrra_amongst_strife_happened
		clr_country_flag = zaradyrra_long_peace_happened
		clr_country_flag = zaradyrra_claw_happened
		clr_country_flag = zaradyrra_blood_canvas_happened
		clr_country_flag = zaradyrra_material_malfeasance_happened
		clr_country_flag = zaradyrra_bad_canvas_happened

		clr_country_flag = zaradyrra_what_become_happened
		clr_country_flag = zaradyrra_band_bards_happened
	}

	option = {
		name = zaradyrra.203.a
		#claw wins
		trigger = {
			check_variable = {
				which = claw_strength
				which = canvas_strength
			}
		}
		#On disaster end, gain permanent modifier, 200 reform progress, and unlock
		# decision to form final tag
		# To get the gov reform, we need to set the appropriate flag, even if you supported the other side
		clr_country_flag = ashhana_disaster_supporting_canvas
		set_country_flag = ashhana_disaster_supporting_claw
		change_government_reform_progress = 200
		add_country_modifier = {
			name = zaradyrra_no_restraints
			duration = -1
		}
	}

	option = {
		name = zaradyrra.203.b
		#canvas wins
		trigger = {
			check_variable = {
				which = canvas_strength
				which = claw_strength
			}
		}
		#On disaster end, gain permanent modifier, 200 reform progress, and unlock
		# decision to form final tag
		# To get the gov reform, we need to set the appropriate flag, even if you supported the other side
		clr_country_flag = ashhana_disaster_supporting_claw
		set_country_flag = ashhana_disaster_supporting_canvas
		change_government_reform_progress = 200

		}
