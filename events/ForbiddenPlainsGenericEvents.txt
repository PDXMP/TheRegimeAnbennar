						name = fp_fauna_rock_crab_vex

country_event = { #Heeding the Headless
	id = fp_flavour.5
	title = fp_flavour.5.t
	desc = fp_flavour.5.d
	picture = TRAILS_BASILISK_eventPicture
	
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = { region = nuzurbokh_region }
				save_event_target_as = event_province
			}
		}
	}

	option = {
		name = fp_flavour.5.a
		event_target:event_province = {
			add_province_modifier = {
				name = fp_fauna_headless_hyena
				duration = 3650 #10 years
			}
		}
	}

	option = {
		name = fp_flavour.5.b
		add_mil_power = -20
		event_target:event_province = {
			add_province_modifier = {
				name = fp_fauna_dire_hyena
				duration = 3650 #10 years
			}
		}
	}

	option = {
		name = fp_flavour.5.c
		if = {
			limit = { monthly_income = 83.333 }
			add_treasury = -100
		}
		else = { add_years_of_income = -0.1 }
		event_target:event_province = {
			if = {
				limit = { development = 25 }
				owner = { add_manpower = -2 }
			}
			else_if = {
				limit = { development = 15 }
				owner = { add_manpower = -1 }
			}
			else_if = {
				limit = { development = 7 }
				owner = { add_manpower = -0.75 }
			}
			else = { owner = { add_manpower = -0.5 } }
			add_province_modifier = {
				name = fp_fauna_headed_off_hyena
				duration = 3650 #10 years
			}
		}
	}
}
