	id = flavour_badakhan.200
	title = flavour_badakhan.200.t
	desc = flavour_badakhan.200.d
	picture = STREET_SPEECH_eventPicture

	hidden = yes
	
	is_triggered_only = yes
	trigger = { has_heir = no }
	
	option = {	#Picking heir
		name = flavour_badakhan.200.a
		random_list = {
			50 = { country_event = { id = flavour_badakhan.201 } }

		}
	}
}
