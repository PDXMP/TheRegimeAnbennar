	desc = flavour_ozgarom.101.d
	picture = DIPLOMACY_eventPicture

	is_triggered_only = yes
	fire_only_once = yes
	hidden = yes

	trigger = {
		owns = 1009
		is_religion_enabled = corinite
		soruin_region = {
			owner = {
				OR = {
					overlord = { capital_scope = { continent = europe } }

				}
			}
		}
		calc_true_if = {
			endralliande_region = {
				type = all
				is_city = yes
				owner = { has_country_modifier = orcish_slaves_in_colonies }
			}
			amount = 10
		}
	}

	immediate = {
		hidden_effect = {
			random_province = {
				limit = {
					region = endralliande_region
					owner = { has_country_modifier = orcish_slaves_in_colonies }
				}
				owner = {
					set_country_flag = H39_bramash_target
				}
			}
		}
	}

	option = {
		name = flavour_ozgarom.101.a
		hidden_effect = {
			every_country = {
				limit = {
					has_discovered = 1009
					OR = {
						has_country_flag = H39_bramash_target
						any_subject_country = { has_country_flag = H39_bramash_target }
						ai = no
					}
				}
				country_event = { id = flavour_ozgarom.102 }
			}
		}
	}
}
