country_event = { #Peace Offer to the Shamakhad
	id = tsukqiao.7
	title = tsukqiao.7.t
	desc = tsukqiao.7.d
	picture = WESTERNISATION_eventPicture

	is_triggered_only = yes
	trigger = { 
		OR = {
			R62 = { exists = yes }
			V98 = { exists = yes }
			AND = {
				has_saved_global_event_target = tsukqiao_hobgobbo_trucymate 
				event_target:tsukqiao_hobgobbo_trucymate = { exists = yes } 
			}
		}
	}

	option = {
		name = tsukqiao.7.a #make truce offer
		trigger = {
			NOT = {
				check_variable = {
					which = TsukqiaoCommandTruceTolerance
					value = 10
				}
			}
		}
		ai_chance = { factor = 100 }
		add_prestige = -10

		if = {
			limit = {
				R62 = { exists = yes } #The Command
			}
			R62 = {
				country_event = {
					id = tsukqiao.8
				}
			}
		}
		else_if = {
			limit = {
				V98 = { exists = yes } #Shammanhome
			}
			V98 = {
				country_event = {
					id = tsukqiao.8
				}
			}
		}
		else = {
			event_target:tsukqiao_hobgobbo_trucymate = {
				country_event = {
					id = tsukqiao.8
				}
			}
		}
		hidden_effect = {
			change_variable = {
				which = TsukqiaoCommandTruceTolerance
				value = 1
			}
		}
	}

	option = {
		name = tsukqiao.7.b #do not make truce offer
		ai_chance = { factor = 0 }
		add_prestige = 15

		if = {
			limit = {
				R62 = { exists = yes } #The Command
			}
			demon_hills_region = { add_permanent_claim = R62 }
		}
		else_if = {
			limit = {
				V98 = { exists = yes } #Shammanhome
			}
			demon_hills_region = { add_permanent_claim = V98 }
		}
		else_if = {
			limit = { has_saved_global_event_target = tsukqiao_hobgobbo_trucymate }
			demon_hills_region = { add_permanent_claim = event_target:tsukqiao_hobgobbo_trucymate }
		}
		clear_global_event_target = tsukqiao_hobgobbo_trucymate
	}
}
