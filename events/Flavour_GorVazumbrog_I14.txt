	id = flavour_gor_vazumbrog.21

	desc = flavour_gor_vazumbrog.21.d
	picture = SCOTTISH_MERCENARIES_eventPicture

	is_triggered_only = yes
	trigger = {
		any_owned_province = {
			region = middle_dwarovar_region
			NOT = { province_group = hold_province }
			is_city = yes
			NOT = { has_fort_building_trigger = yes }
			has_tax_building_trigger = yes
		}
	}

	immediate = {
		random_owned_province = {
			limit = {
				region = middle_dwarovar_region
				NOT = { province_group = hold_province }
				is_city = yes
				NOT = { has_fort_building_trigger = yes }
				has_tax_building_trigger = yes
			}
			save_event_target_as = dragon_attack_holy_site
		}
	}
	goto = dragon_attack_holy_site

	option = {
		name = flavour_gor_vazumbrog.21.a
		if = {
			limit = { stability = -2 }
			add_stability = -1
		}
		event_target:dragon_attack_holy_site = {
			if = {
				limit = { has_building = temple }
				remove_building = temple
			}
			else = {
				remove_building = cathedral
			}
			add_devastation = 70
		}
	}
}

					change_trade_goods = serpentbloom
				}
				owner = { clr_country_flag = vault_hearthstone_engine_enabled } # end of event chain - until fired again from province button
			}
		}
		else = {
			owner = { clr_country_flag = vault_hearthstone_engine_enabled } # end of event chain - until fired again from province button
		}
	}
