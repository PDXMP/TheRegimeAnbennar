		}

	}
}

#Unlocking cults from subject/overlord
country_event = {
	id = goblinic_shamanism_flavor.2
	title = goblinic_shamanism_flavor.2.t #Never Seen
	desc = goblinic_shamanism_flavor.2.t #Never Seen
	picture = POPE_PREACHING_eventPicture #Never Seen
	
	hidden = yes
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = {
				is_subject = yes
				overlord = {
					goblinic_shamanism_trigger = yes
				}
			}
			overlord = {
				goblinic_shamanism_event = yes
			}
		}
		else = {
			random_subject_country = {
				limit = {
					goblinic_shamanism_trigger = yes
				}
				goblinic_shamanism_event = yes
			}
		}
	}

	trigger = {
		religion = goblinic_shamanism
		has_dlc = "Rights of Man"
		OR = {
			AND = {
				is_subject = yes
				overlord = {
					goblinic_shamanism_trigger = yes
				}
			}
			any_subject_country = {
				goblinic_shamanism_trigger = yes
			}
		}
	}

	option = {
		name = goblinic_shamanism_flavor.2.t #Never seen
	}
}

	desc = "goblinic_shamanism_flavor.257.d"
	picture = FETISHIST_FIRE_eventPicture

	trigger = {
		has_adopted_cult = centaur_cult
		any_owned_province = {
			base_tax = 2
			base_production = 2
			base_manpower = 2
		}
	}

	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					base_tax = 2
					base_production = 2
					base_manpower = 2
				}
				save_event_target_as = goblinic_shamanism_raze_province
			}
		}
	}
	
	goto = goblinic_shamanism_raze_province
	
	# Burn it all!
	option = {
		name = "goblinic_shamanism_flavor.257.a"
		add_adm_power = 50
		add_dip_power = 50
		add_mil_power = 50
		event_target:goblinic_shamanism_raze_province = {
			add_base_tax = -1
			add_base_production = -1
			add_base_manpower = -1
			add_unrest = 5
		}
	}
}
