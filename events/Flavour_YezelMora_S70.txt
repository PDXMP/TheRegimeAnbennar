		# This is just how we make say "it's better" vs "it's worse"
		trigger = {
			NOT = {
				is_year = 1703
			}
		}
		desc = shadow.8.desc.a
	}
	
	desc = {
		trigger = {
			is_year = 1703
		}
		desc = shadow.8.desc.b
	}

	picture = MAGE_MOON_eventPicture

	# This shows when the alignment is more favourable

		name = shadow.8.a
