	title = flavour_great_storm.73.t
	desc = flavour_great_storm.73.d1

	picture = ARTIFICE_GEAR_eventPicture
	
	is_triggered_only = yes
	hidden = yes

	immediate = {
		if = {
			limit = { has_cast_elemental_fury = yes } 
			increase_magic_experience = { unmodified_amount = 100 } 
		}
		else_if = {
			limit = {
				OR = {
					has_cast_mage_armor = yes
					has_cast_manipulated_fortune = yes
					has_cast_mass_charm_2 = yes					
				}
			}
			increase_magic_experience = { unmodified_amount = 50 } 
		}
		else_if = {
			limit = {
				OR = {
					has_cast_mass_charm = yes
					has_cast_shock_and_awe = yes
					has_cast_flaming_munitions = yes
				}
			}
			increase_magic_experience = { unmodified_amount = 25 } 
		}
		else = {
			increase_magic_experience = { unmodified_amount = 12.5 }
		}
	}

	option = {
		name = flavour_great_storm.73.a
		
		change_variable = { magic_projects_completed = 1 }
	}
}
